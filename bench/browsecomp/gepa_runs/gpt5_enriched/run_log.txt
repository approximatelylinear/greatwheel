def norm_name(s):
    if not s: return s
    # remove honorifics/suffixes; drop middle initials if present
    drop = {"mr", "mrs", "ms", "dr", "sir", "dame", "prof", "jr", "sr"}
    parts = [p for p in s.replace(".", "").split() if p.lower() not in drop]
    # drop single-letter middle tokens
    cleaned = [p for p in parts if len(p) > 1 or p.lower() in {"van","von","de","da","di","del","la","le","du"}]

    entry["answers"].add(ans)
    entry["evidence"].append(ev)
    entry["docids"].add(docid)
    # Score: doc support + confidence
    entry["score"] += 0.6*conf + 0.4
